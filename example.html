    autoplay: true,
    showinfo: true
  });
});

//Javascript:
$('#input').blur(function(){
  var html = $.youtube.element('#element', $('#input').val());

  $('#container').html(html);

  $('#element').youtube('video', {
    autoplay: true,
    width:    '300',
    height:   '250'
  });
});

/*
 * Html:
 * <input type="text" id="input">
 *
 * <center id="container"></center>
